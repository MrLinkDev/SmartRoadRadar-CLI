33 frame read_frame() {
34 frame received_frame{};
35
36 while (data_bus.read_u_byte() != HEADER_DATA_FRAME_1);
37
38 if (data_bus.read_u_byte() != HEADER_DATA_FRAME_2) {
39 received_frame.is_valid = false;
40 return received_frame;
41 }
42
43 received_frame.data_length.b[0] = data_bus.read_u_byte();
44 received_frame.data_length.b[1] = data_bus.read_u_byte();
45
46 received_frame.word = data_bus.read_u_byte();
47
48 if (received_frame.data_length.s > 1) {
49 received_frame.data = new u_byte_t[received_frame.data_length.s - 1];
50
51 data_bus.read_u_bytes(received_frame.data, received_frame.data_length.s - 1);
52 }
53
54 received_frame.checksum = data_bus.read_u_byte();
55
56 u_byte_t calc_checksum = calculate_checksum(received_frame);
57 if (calc_checksum == received_frame.checksum) {

61 }

63 return received_frame;

111 int write_frame(frame target_frame) {
112 int packet_length =
113 LENGTH_HEADER +
114 LENGTH_DATA_LENGTH +
115 LENGTH_COMMAND_WORD +
116 (target_frame.data_length.s - 1) +
117 LENGTH_CHECKSUM;
118
119 u_byte_t packet[packet_length];
120
121 packet[0] = target_frame.header[0];
122 packet[1] = target_frame.header[1];
123
124 packet[2] = target_frame.data_length.b[0];
125 packet[3] = target_frame.data_length.b[1];
126
127 packet[4] = target_frame.word;
128
129 if (target_frame.data != nullptr) {
130 for (int i = 5; i < packet_length - 1; ++i) {
131 packet[i] = target_frame.data[i - 5];
132 }
133 }
134
135 packet[packet_length - 1] = target_frame.checksum;
136
137 int result = data_bus.write_u_bytes(packet, packet_length);
138 return result;
139 }

462 int get_target_data(target_data *data) {
463 frame received_frame;
464
465 do {
466 received_frame = read_expected_frame(CMD_READ_TARGET_DATA);
467 } while (received_frame.data_length.s <= 1);
468
469 if (received_frame.is_valid) {
470 int target_count = (received_frame.data_length.s - 1) / TARGET_DATA_BYTE_LENGTH;
471
472 for (int pos = 0; pos < target_count; ++pos) {
473 data[pos].num = received_frame.data[0 + TARGET_DATA_BYTE_LENGTH * pos];
474
475 data[pos].distance = u_byte_to_float(new u_byte_t[2] {
476 received_frame.data[1 + TARGET_DATA_BYTE_LENGTH * pos],
477 received_frame.data[2 + TARGET_DATA_BYTE_LENGTH * pos]
478 });
479
480 data[pos].speed = u_byte_to_float(new u_byte_t[2] {
481 received_frame.data[3 + TARGET_DATA_BYTE_LENGTH * pos],
482 received_frame.data[4 + TARGET_DATA_BYTE_LENGTH * pos]
483 });
484
485 data[pos].angle = u_byte_to_float(new u_byte_t[2] {
486 received_frame.data[5 + TARGET_DATA_BYTE_LENGTH * pos],
487 received_frame.data[6 + TARGET_DATA_BYTE_LENGTH * pos]
488 });
489
490 data[pos].snr =u_byte_to_float(new u_byte_t[2] {
491 received_frame.data[7 + TARGET_DATA_BYTE_LENGTH * pos],
492 received_frame.data[8 + TARGET_DATA_BYTE_LENGTH * pos]
