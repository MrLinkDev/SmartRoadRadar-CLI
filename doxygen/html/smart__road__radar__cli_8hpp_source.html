62 void parse_line(std::string *line) {

70 if (cmd == CLI_VERSION || cmd == CLI_VERSION_SHORT) {

72 get_version();
73 else
74 usage();
75 } else if (cmd == CLI_SET_PARAMS || cmd == CLI_SET_PARAMS_SHORT) {
76 if (line->length() > 0)

95 } else if (cmd == CLI_ENABLE_TRANSMIT || cmd == CLI_ENABLE_TRANSMIT_SHORT) {

103 else
104 usage();
105 } else if (cmd == CLI_SET_DATA_FREQ || cmd == CLI_SET_DATA_FREQ_SHORT) {
106 if (line->length() > 0)
107 set_data_freq(*line);
108 else
109 usage();
110 } else if (cmd == CLI_ENABLE_ZERO_DATA || cmd == CLI_ENABLE_ZERO_DATA_SHORT) {

121 usage();
