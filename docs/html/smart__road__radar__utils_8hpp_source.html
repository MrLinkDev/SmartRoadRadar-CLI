195u_byte_t calculate_checksum(frame target_frame) {
196 u_byte_t checksum = 0x00;
197
198 checksum += target_frame.data_length.b[0];
199 checksum += target_frame.data_length.b[1];
200
201 checksum += target_frame.word;
202
203 if (target_frame.data_length.s > 1) {
204 for (unsigned short i = 0; i < target_frame.data_length.s - 1; ++i) {
205 checksum += target_frame.data[i];
206 }
207 }
208
209 return checksum;
210}
